    <script>
        let watchId = null;

        function requestGpsPermission() {
            const statusMsg = document.getElementById('status-msg');
            const startBtn = document.getElementById('start-btn');

            // Kiểm tra trình duyệt có hỗ trợ GPS không
            if (!navigator.geolocation) {
                statusMsg.innerHTML = "<span style='color: #ff4a4a;'>Trình duyệt của bạn không hỗ trợ định vị GPS định sẵn.</span>";
                return;
            }

            statusMsg.innerText = "Đang gửi yêu cầu cấp quyền từ hệ thống...";

            // Cấu hình GPS chuyên dụng cho việc đo tốc độ di chuyển
            const gpsOptions = {
                enableHighAccuracy: true, // Bắt buộc bật định vị chính xác cao (Dùng chip GPS thay vì Wifi/Mạng di động)
                timeout: 10000,           // Thời gian tối đa chờ phản hồi vệ tinh (10 giây)
                maximumAge: 0             // Không sử dụng dữ liệu vị trí cũ lưu trong bộ nhớ đệm
            };

            // Tiến hành theo dõi vị trí liên tục (Trình duyệt sẽ hiện bảng thông báo cho phép/chặn tại đây)
            watchId = navigator.geolocation.watchPosition(updateSpeedometer, handleGpsError, gpsOptions);
            
            // Đổi trạng thái nút bấm
            startBtn.innerText = "ĐANG ĐO TỐC ĐỘ...";
            startBtn.classList.add('active-gps');
            startBtn.disabled = true;
        }

        // HÀM XỬ LÝ DỮ LIỆU KHI THIẾT BỊ DI CHUYỂN
        function updateSpeedometer(position) {
            const coords = position.coords;
            
            // 1. Cập nhật Kinh độ và Vĩ độ lên màn hình
            document.getElementById('lat-val').innerText = coords.latitude.toFixed(6);
            document.getElementById('lng-val').innerText = coords.longitude.toFixed(6);
            
            // Hiển thị độ sai số bán kính (để biết sóng GPS có khỏe hay không)
            document.getElementById('accuracy-val').innerText = `Độ sai số GPS: ±${coords.accuracy.toFixed(1)} mét`;

            // 2. Lấy tốc độ trả về từ phần cứng GPS máy (m/s)
            let speedInMeterPerSecond = coords.speed;

            // Kiểm tra xem dữ liệu tốc độ trả về có hợp lệ không (Nếu đứng im hoặc sóng yếu, giá trị có thể bị null hoặc âm)
            if (speedInMeterPerSecond === null || speedInMeterPerSecond < 0 || isNaN(speedInMeterPerSecond)) {
                document.getElementById('speed-text').innerText = "0";
                document.getElementById('status-msg').innerText = "Đã kết nối vệ tinh. Hãy di chuyển để bắt đầu đo.";
            } else {
                // Đổi đơn vị từ m/s sang Km/h (Nhân với hệ số 3.6)
                let speedInKmH = speedInMeterPerSecond * 3.6;
                
                // Làm tròn số nguyên hiển thị giống xe máy/ô tô
                document.getElementById('speed-text').innerText = Math.round(speedInKmH);
                document.getElementById('status-msg').innerText = "Tín hiệu GPS tốt! Hệ thống đang cập nhật.";
            }
        }

        // HÀM XỬ LÝ LỖI (Khi người dùng bấm Từ chối hoặc mất sóng)
        function handleGpsError(error) {
            const statusMsg = document.getElementById('status-msg');
            const startBtn = document.getElementById('start-btn');
            
            // Reset lại trạng thái nút bấm để người dùng có thể bấm lại
            startBtn.innerText = "THỬ LẠI KẾT NỐI";
            startBtn.classList.remove('active-gps');
            startBtn.disabled = false;

            switch(error.code) {
                case error.PERMISSION_DENIED:
                    statusMsg.innerHTML = "<span style='color: #ff4a4a;'>Bạn đã TỪ CHỐI cấp quyền vị trí. Hãy vào cài đặt trình duyệt để mở lại quyền cho trang web này.</span>";
                    break;
                case error.POSITION_UNAVAILABLE:
                    statusMsg.innerHTML = "<span style='color: #ffbc4a;'>Không thể xác định vị trí. Vui lòng bật định vị hệ thống (Location/GPS) trên điện thoại của bạn.</span>";
                    break;
                case error.TIMEOUT:
                    statusMsg.innerHTML = "<span style='color: #ffbc4a;'>Hết thời gian tìm kiếm vệ tinh. Hãy di chuyển ra vùng không gian thoáng đãng hơn.</span>";
                    break;
                default:
                    statusMsg.innerHTML = "<span style='color: #ff4a4a;'>Đã xảy ra lỗi hệ thống không xác định.</span>";
                    break;
            }
        }
